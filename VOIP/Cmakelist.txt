cmake_minimum_required(VERSION 3.10)
project(VoipFltk)

set(CMAKE_CXX_STANDARD 17)

find_package(FLTK REQUIRED)
find_package(PkgConfig REQUIRED)
pkg_check_modules(PORTAUDIO REQUIRED portaudio-2.0)

include_directories(${FLTK_INCLUDE_DIR} ${PORTAUDIO_INCLUDE_DIRS})

add_executable(voip-fltk
    main.cpp
    voip_core.cpp
    audio_io.cpp
    jitter_buffer.cpp
)

target_link_libraries(voip-fltk
    ${FLTK_LIBRARIES}
    ${PORTAUDIO_LIBRARIES}
    pthread
)
